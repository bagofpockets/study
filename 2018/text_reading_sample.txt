#include "stdafx.h"
#include <iostream>
#include <string>
#include <fstream>
#include <sstream>
using namespace std;

namespace
{
	ifstream infile("text_sample.txt");
}

void token_based()
{
	int a, b;
	while (infile >> a >> b)
	{
		cout << a << " " << b << "\n";
	}
}

void line_based()
{
	string line;
	while (getline(infile, line))
	{
		istringstream iss(line);
		int a, b;
		if (!(iss >> a >> b)) { break; } // error
		cout << a << " " << b << "\n";					
	}
}

int main()
{
	int a;
	cin >> a;
	if (a == 1) 
	{ 
		token_based(); 
	}
	else
	{
		line_based();
	}
	system("pause");
    return 0;
}
